Kruskal's Algorithm Code

class ResourceSharingManager {
private:
    vector> edges;
    vector parent;
    vector rank;

    int find(int node) {
        if (parent[node] != node)
            parent[node] = find(parent[node]);
        return parent[node];
    }

    void unite(int u, int v) {
        int rootU = find(u);
        int rootV = find(v);
        if (rootU != rootV) {
            if (rank[rootU] > rank[rootV]) {
                parent[rootV] = rootU;
            } else if (rank[rootU] < rank[rootV]) {
                parent[rootU] = rootV;
            } else {
                parent[rootV] = rootU;
                rank[rootU]++;
            }
        }
    }

public:
    ResourceSharingManager(int n) : parent(n), rank(n, 0) {
        for (int i = 0; i < n; ++i)
            parent[i] = i;
    }

    void addEdge(int u, int v, int weight) {
        edges.emplace_back(u, v, weight);
    }

    void computeMST() {
        sort(edges.begin(), edges.end(), [](const auto& a, const auto& b) {
            return get<2>(a) < get<2>(b);
        });

        cout << "Kruskal's MST:\n";
        int mstWeight = 0;
        for (const auto& [u, v, weight] : edges) {
            if (find(u) != find(v)) {
                cout << u << " - " << v << " (Weight: " << weight << ")\n";
                unite(u, v);
                mstWeight += weight;
            }
        }
        cout << "Total Weight of MST: " << mstWeight << "\n";
    }
};
