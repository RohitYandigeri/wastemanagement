class WasteClassifier {
private:
    struct WasteCategory {
        string type;
        string process;

        WasteCategory(string type, string process) : type(type), process(process) {}
    };

    vector categories;

public:
    void addCategories() {
        int n;
        cout << "Enter the number of waste categories: ";
        cin >> n;

        for (int i = 0; i < n; ++i) {
            string type, process;
            cout << "Enter Waste Type and Process: ";
            cin >> type >> process;
            categories.emplace_back(type, process);
        }
    }

    void classifyWaste() {
        sort(categories.begin(), categories.end(), [](const WasteCategory& a, const WasteCategory& b) {
            return a.type < b.type;
        });

        cout << "Sorted Waste Categories:\n";
        for (const auto& category : categories) {
            cout << "Type: " << category.type << ", Process: " << category.process << "\n";
        }

        string query;
        cout << "Enter the waste type to classify: ";
        cin >> query;

        auto it = find_if(categories.begin(), categories.end(), [&](const WasteCategory& c) {
            return c.type == query;
        });

        if (it != categories.end()) {
            cout << "Query Result: Waste Type: " << query << " -> Process: " << it->process << "\n";
        } else {
            cout << "Query Result: Waste Type " << query << " not found in categories.\n";
        }
    }
};
