class DigitalWasteManager {
private:
    struct Interaction {
        int userId;
        int locationId;
        string issue;

        Interaction(int userId, int locationId, string issue)
            : userId(userId), locationId(locationId), issue(issue) {}
    };

    vector interactions;

public:
    void addInteraction() {
        int userId, locationId;
        string issue;
        cout << "Enter User ID, Location ID, and Issue: ";
        cin >> userId >> locationId >> issue;
        interactions.emplace_back(userId, locationId, issue);
    }

    void displayInteractions() {
        unordered_map>> departmentMap;

        for (const auto& interaction : interactions) {
            departmentMap[interaction.issue].emplace_back(interaction.userId, interaction.locationId);
        }

        cout << "Reported Issues:\n";
        for (const auto& entry : departmentMap) {
            cout << "Issue: " << entry.first << "\n";
            for (const auto& [userId, locationId] : entry.second) {
                cout << "  User ID: " << userId << ", Location ID: " << locationId << "\n";
            }
        }
    }
};
