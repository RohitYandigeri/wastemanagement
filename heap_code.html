using namespace std;

class BinManager {
private:
    struct Bin {
        int id;
        int fillLevel;
        vector history;

        Bin(int id, int fillLevel, vector history)
            : id(id), fillLevel(fillLevel), history(history) {}
    };

    static int predictFillLevel(const vector& history) {
        int n = history.size();
        return n < 2 ? history.back() : history.back() + (history.back() - history[n - 2]);
    }

public:
    void sortAndPredictBins() {
        int n;
        cout << "Enter the number of bins: ";
        cin >> n;

        vector bins;
        for (int i = 0; i < n; ++i) {
            int id, fillLevel, historySize;
            cout << "Enter Bin ID and Fill Level: ";
            cin >> id >> fillLevel;
            cout << "Enter number of history records: ";
            cin >> historySize;

            vector history(historySize);
            cout << "Enter history records: ";
            for (int j = 0; j < historySize; ++j) cin >> history[j];

            bins.emplace_back(id, fillLevel, history);
        }

        auto compare = [](const Bin& a, const Bin& b) {
            return a.fillLevel < b.fillLevel;
        };

        priority_queue, decltype(compare)> pq(compare);
        for (const auto& bin : bins) pq.push(bin);

        cout << "Bins sorted by fill level with predictions:\n";
        while (!pq.empty()) {
            auto bin = pq.top();
            pq.pop();
            cout << "Bin ID: " << bin.id << ", Fill Level: " << bin.fillLevel
                 << ", Predicted: " << predictFillLevel(bin.history) << "\n";
        }
    }
};
