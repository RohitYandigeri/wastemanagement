class HazardousWasteDisposal {
private:
    vector> graph;

public:
    void initializeGraph() {
        int n;
        cout << "Enter the number of nodes: ";
        cin >> n;

        graph.resize(n, vector(n, 0));
        cout << "Enter the adjacency matrix (" << n << "x" << n << "):\n";
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < n; ++j)
                cin >> graph[i][j];
    }

    void primsAlgorithm() {
        int n = graph.size();
        vector key(n, INT_MAX);
        vector mstSet(n, false);
        vector parent(n, -1);
        key[0] = 0;

        for (int count = 0; count < n - 1; ++count) {
            int minKey = INT_MAX, u = -1;
            for (int v = 0; v < n; ++v) {
                if (!mstSet[v] && key[v] < minKey) {
                    minKey = key[v];
                    u = v;
                }
            }

            mstSet[u] = true;

            for (int v = 0; v < n; ++v) {
                if (graph[u][v] && !mstSet[v] && graph[u][v] < key[v]) {
                    key[v] = graph[u][v];
                    parent[v] = u;
                }
            }
        }

        cout << "Prim's MST:\n";
        for (int i = 1; i < n; ++i) {
            if (parent[i] != -1) {
                cout << parent[i] << " - " << i << " (Weight: " << graph[i][parent[i]] << ")\n";
            }
        }
    }
}
