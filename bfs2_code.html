class SewagePathExplorer {
private:
    vector> graph;

public:
    void initializeGraph() {
        int n;
        cout << "Enter the number of nodes: ";
        cin >> n;

        graph.resize(n, vector(n, 0));
        cout << "Enter the adjacency matrix (" << n << "x" << n << "):\n";
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < n; ++j)
                cin >> graph[i][j];
    }

    void bfsTraversal(int startNode) {
        vector visited(graph.size(), false);
        queue q;

        q.push(startNode);
        visited[startNode] = true;

        cout << "BFS Traversal from Node " << startNode << ": ";
        while (!q.empty()) {
            int currentNode = q.front();
            q.pop();

            cout << currentNode << " ";

            for (int i = 0; i < graph[currentNode].size(); ++i) {
                if (graph[currentNode][i] && !visited[i]) {
                    q.push(i);
                    visited[i] = true;
                }
            }
        }
        cout << "\n";
    }
};
